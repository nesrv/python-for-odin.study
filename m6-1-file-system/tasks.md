### **–ó–∞–¥–∞–Ω–∏–µ 3. –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –ø–æ–¥—Å—á—ë—Ç —Ñ–∞–π–ª–æ–≤ –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏**
**–£—Å–ª–æ–≤–∏–µ:**  
–ù–∞–ø–∏—à–∏—Ç–µ —Å–∫—Ä–∏–ø—Ç, –∫–æ—Ç–æ—Ä—ã–π —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ –æ–±—Ö–æ–¥–∏—Ç —É–∫–∞–∑–∞–Ω–Ω—É—é –ø–∞–ø–∫—É –∏ –≤—ã–≤–æ–¥–∏—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–∞–π–ª–æ–≤ –≤ –Ω–µ–π (–∏ –≤–æ –≤—Å–µ—Ö –ø–æ–¥–ø–∞–ø–∫–∞—Ö).

### **–ó–∞–¥–∞–Ω–∏–µ 4. –ö–æ–ø–∏—Ä–æ–≤–∞–Ω–∏–µ –≤—Å–µ—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –≤ –æ—Ç–¥–µ–ª—å–Ω—É—é –ø–∞–ø–∫—É**
**–£—Å–ª–æ–≤–∏–µ:**  
–°–æ–∑–¥–∞–π—Ç–µ –ø–∞–ø–∫—É `images`, –∑–∞—Ç–µ–º —Å–∫–æ–ø–∏—Ä—É–π—Ç–µ –≤ –Ω–µ—ë –≤—Å–µ —Ñ–∞–π–ª—ã —Å —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è–º–∏ `.jpg`, `.png` –∏ `.gif` –∏–∑ —Ç–µ–∫—É—â–µ–π –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏.

**–†–µ—à–µ–Ω–∏–µ:**

import os
import shutil

# –°–æ–∑–¥–∞–µ–º –ø–∞–ø–∫—É images, –µ—Å–ª–∏ –µ—ë –Ω–µ—Ç

    os.mkdir("images")

# –ö–æ–ø–∏—Ä—É–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è

    if filename.lower().endswith((".jpg", ".png", ".gif")):

### **–ó–∞–¥–∞–Ω–∏–µ 5. –ü–æ–∏—Å–∫ —Å–∞–º–æ–≥–æ –±–æ–ª—å—à–æ–≥–æ —Ñ–∞–π–ª–∞ –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏**
**–£—Å–ª–æ–≤–∏–µ:**  
–ù–∞–π–¥–∏—Ç–µ —Ñ–∞–π–ª —Å –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–º —Ä–∞–∑–º–µ—Ä–æ–º –≤ —É–∫–∞–∑–∞–Ω–Ω–æ–π –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ (–∏ –µ—ë –ø–æ–¥–ø–∞–ø–∫–∞—Ö) –∏ –≤—ã–≤–µ–¥–∏—Ç–µ –µ–≥–æ –∏–º—è –∏ —Ä–∞–∑–º–µ—Ä –≤ –º–µ–≥–∞–±–∞–π—Ç–∞—Ö.

**–†–µ—à–µ–Ω–∏–µ:**
```python
import os

def find_largest_file(directory):
    max_size = 0
    largest_file = ""

    for root, dirs, files in os.walk(directory):
        for file in files:
            file_path = os.path.join(root, file)
            size = os.path.getsize(file_path)
            if size > max_size:
                max_size = size
                largest_file = file_path

    return largest_file, max_size / (1024 * 1024)  # –†–∞–∑–º–µ—Ä –≤ –ú–ë

path = input("–í–≤–µ–¥–∏—Ç–µ –ø—É—Ç—å –∫ –ø–∞–ø–∫–µ: ")
file, size = find_largest_file(path)
print(f"–°–∞–º—ã–π –±–æ–ª—å—à–æ–π —Ñ–∞–π–ª: {file}, —Ä–∞–∑–º–µ—Ä: {size:.2f} –ú–ë")
```
